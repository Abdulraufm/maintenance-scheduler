<script>

// EmailJS initialization

emailjs.init('YOUR_USER_ID'); // Replace with your EmailJS user ID


// 1. Set Maintenance Repair Date and Time

function setRepairDate() {

const repairDate = document.getElementById('repairDate').value;

localStorage.setItem('repairDate', repairDate); // Store in local storage

document.getElementById('repairDetails').innerText = `Next Maintenance Repair: ${repairDate}`;

}


// Load saved repair date when the page is refreshed

window.onload = function() {

const savedDate = localStorage.getItem('repairDate');

if (savedDate) {

document.getElementById('repairDetails').innerText = `Next Maintenance Repair: ${savedDate}`;

}

loadAssets(); // Load asset list when the page is loaded

};


// 2. Schedule Servicing Reminder

function scheduleReminder() {

const servicingDate = document.getElementById('servicingDate').value;

const emailData = {

to_email: 'recipient@example.com', // Replace with the user's email

servicing_date: servicingDate

};


// Send email reminder using EmailJS

emailjs.send('YOUR_SERVICE_ID', 'YOUR_TEMPLATE_ID', emailData)

.then((response) => {

alert('Reminder scheduled successfully!');

}, (error) => {

alert('Failed to send reminder: ' + error.text);

});

}


// 3. My Asset List

function loadAssets() {

const assets = JSON.parse(localStorage.getItem('assets')) || [];

const assetList = document.getElementById('assetList');

assetList.innerHTML = '';

assets.forEach((asset, index) => {

const li = document.createElement('li');

li.innerText = asset;

assetList.appendChild(li);

});

}


function addAsset() {

const assetName = document.getElementById('assetName').value;

if (assetName) {

const assets = JSON.parse(localStorage.getItem('assets')) || [];

assets.push(assetName);

localStorage.setItem('assets', JSON.stringify(assets));

loadAssets();

document.getElementById('assetName').value = ''; // Clear input field

}

}


// 4. Emergency Repair Request

function sendEmergencyRepair() {

const issueDetails = document.getElementById('repairDetailsText').value;

const emailData = {

to_email: 'manager@example.com', // Replace with manager's email

message: issueDetails

};


emailjs.send('YOUR_SERVICE_ID', 'YOUR_TEMPLATE_ID', emailData)

.then((response) => {

alert('Emergency repair request sent successfully!');

}, (error) => {

alert('Failed to send request: ' + error.text);

});

}
